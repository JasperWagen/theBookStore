@app.route('/api/v1/resources/wwe', methods=['GET'])
def api_filter():
    query_parameters = request.args
    # id = query_parameters.get('id')
    name = query_parameters.get('name')
    gender = query_parameters.get('gender')
    brand = query_parameters.get('brand')

    query = "SELECT * FROM starsdata WHERE"
    to_filter = []

    # if id:
    #     query += ' id=? AND'
    #     to_filter.append(id)
    if name:
        query += ' name=? AND'
        to_filter.append(name)
    if gender:
        query += ' gender=? AND'
        to_filter.append(gender)
    if brand:
        query += ' brand=? AND'
        to_filter.append(brand)
    #if not (id or name or gender or brand):
    if not (name or gender or brand):
        return page_not_found(404)

    query = query[:-4] + ';'

    conn = sqlite3.connect('wwedatabase')
    conn.row_factory = dict_factory
    cur = conn.cursor()

    results = cur.execute(query, to_filter).fetchall()

    return jsonify(results)
